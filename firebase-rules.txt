// Regles Firestore pour VIRUS
// A copier dans Firebase Console > Firestore > Rules
//
// NOTE: Ces regles ajoutent une validation basique des donnees.
// Sans Firebase Auth, on ne peut pas verifier l'identite du joueur,
// mais on peut valider le format et la taille des donnees.

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Players : lecture publique, ecriture avec validation
    match /players/{playerId} {
      allow read: if true;
      allow create: if request.resource.data.keys().hasAll(['playerId', 'pseudo'])
        && request.resource.data.pseudo is string
        && request.resource.data.pseudo.size() >= 2
        && request.resource.data.pseudo.size() <= 20;
      allow update: if request.resource.data.pseudo is string
        && request.resource.data.pseudo.size() <= 20
        && (!request.resource.data.keys().hasAny(['gold']) || request.resource.data.gold is number)
        && (!request.resource.data.keys().hasAny(['gold']) || request.resource.data.gold >= 0)
        && (!request.resource.data.keys().hasAny(['gold']) || request.resource.data.gold <= 1000000)
        && (!request.resource.data.keys().hasAny(['wins']) || request.resource.data.wins is number)
        && (!request.resource.data.keys().hasAny(['wins']) || request.resource.data.wins >= 0)
        && (!request.resource.data.keys().hasAny(['kills']) || request.resource.data.kills is number)
        && (!request.resource.data.keys().hasAny(['kills']) || request.resource.data.kills >= 0)
        && (!request.resource.data.keys().hasAny(['gamesPlayed']) || request.resource.data.gamesPlayed is number)
        && (!request.resource.data.keys().hasAny(['gamesPlayed']) || request.resource.data.gamesPlayed >= 0)
        && (!request.resource.data.keys().hasAny(['deaths']) || request.resource.data.deaths is number)
        && (!request.resource.data.keys().hasAny(['deaths']) || request.resource.data.deaths >= 0)
        && (!request.resource.data.keys().hasAny(['xp']) || request.resource.data.xp is number)
        && (!request.resource.data.keys().hasAny(['xp']) || request.resource.data.xp >= 0)
        && (!request.resource.data.keys().hasAny(['level']) || request.resource.data.level is number)
        && (!request.resource.data.keys().hasAny(['level']) || request.resource.data.level >= 1)
        && (!request.resource.data.keys().hasAny(['pet']) || request.resource.data.pet is string)
        && (!request.resource.data.keys().hasAny(['pet']) || request.resource.data.pet.size() <= 50)
        && (!request.resource.data.keys().hasAny(['petsAchetes']) || request.resource.data.petsAchetes is list);
      allow delete: if true;
    }

    // Parties : lecture publique, creation/modification avec validation
    match /parties/{partyId} {
      allow read: if true;
      allow write: if true;
    }

    // PartyPlayers : lecture publique, ecriture avec validation pseudo
    match /partyPlayers/{docId} {
      allow read: if true;
      allow create: if request.resource.data.pseudo is string
        && request.resource.data.pseudo.size() <= 20;
      allow update, delete: if true;
    }

    // Chat messages : validation taille message
    match /chatMessages/{docId} {
      allow read: if true;
      allow create: if request.resource.data.message is string
        && request.resource.data.message.size() <= 200
        && request.resource.data.pseudo is string
        && request.resource.data.pseudo.size() <= 20;
      allow update, delete: if false;
    }

    // Cadavres (corps des joueurs elimines)
    match /cadavres/{docId} {
      allow read: if true;
      allow create: if true;
      allow update, delete: if true;
    }

    // Meetings (reunions d'urgence)
    match /meetings/{docId} {
      allow read: if true;
      allow write: if true;
    }

    // Votes : creation uniquement (pas de modification)
    match /votes/{docId} {
      allow read: if true;
      allow create: if request.resource.data.keys().hasAll(['partyId', 'meetingId', 'voterId'])
        && request.resource.data.voterPseudo is string
        && request.resource.data.voterPseudo.size() <= 20;
      allow update, delete: if false;
    }

    // Friend requests
    match /friendRequests/{docId} {
      allow read: if true;
      allow write: if true;
    }

    // Friends (relations bidirectionnelles)
    match /friends/{docId} {
      allow read: if true;
      allow write: if true;
    }

    // Invitations de partie
    match /gameInvites/{docId} {
      allow read: if true;
      allow write: if true;
    }

    // Version du jeu
    match /version/{docId} {
      allow read: if true;
      allow write: if false;
    }

    // Echanges de skins
    match /skinTrades/{docId} {
      allow read: if true;
      allow write: if true;
    }

    // Signalements
    match /signalements/{docId} {
      allow read: if true;
      allow create: if true;
      allow update, delete: if false;
    }

    // Missions completees
    match /missionsCompletees/{docId} {
      allow read: if true;
      allow create: if true;
      allow update, delete: if false;
    }

    // Reports (bugs / signalements) — envoyes par joueurs, vus par admins dev
    match /reports/{docId} {
      allow read: if true;
      allow create: if request.resource.data.message is string
        && request.resource.data.message.size() <= 500
        && request.resource.data.pseudo is string
        && request.resource.data.pseudo.size() <= 20
        && request.resource.data.type is string
        && (request.resource.data.type == 'bug' || request.resource.data.type == 'signalement');
      // update (pour marquer 'checkedAt') et delete (cleanup auto + suppr manuelle admin)
      allow update, delete: if true;
    }

    // Feature flags (admin dev contrôle l'activation des features)
    match /featureFlags/{flagId} {
      allow read: if true;
      allow write: if true;
    }

    // Salon groups : groupes d'amis dans le salon
    match /salonGroups/{groupId} {
      allow read: if true;
      allow write: if true;
    }

    // Invitations a un salon group
    match /salonGroupInvites/{inviteId} {
      allow read: if true;
      allow write: if true;
    }
  }
}
